void print(float*m, int n)
{
	for (int i = 0; i < n; i++)
	{
		cout << m[i] <<" ";
	}
	cout << endl;
}
void fill(float*m,int n)
{
	for (int i = 0; i < n;i++)
	{
		m[i] = (rand() & 100 - 50)/10;

	}
}
void sort(float*m, int n)
{
	for (int i = 0; i < n;i++)
	{
		for (int i = 0; i < n-1;i++)
		{
			if (m[i] > m[i + 1])
			{
				swap(m[i], m[i + 1]);
			}
		}
	}	
}
float max(float* m, int n)
{
	int max = m[0];
	for (int i = 0; i < n; i++)
	{
		if (m[i] > max)
		{
			max = m[i];
		}
	}
	return max;
}
